```py

import gspread
import random
import numpy as np
gc = gspread.service_account(filename='unitydatascience-400614-72a1d5be7975.json')
sh = gc.open("Homework2")
mon = list(range(1,100))
success = False
count = 0
chance = 1

while not success:
    sh.sheet1.clear()
    count += 1
    if random.random() < chance/100:
        success = True
```
